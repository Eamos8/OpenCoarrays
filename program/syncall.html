program syncall
  implicit none

  integer :: me,np,i
  integer :: scalar[*]

  me = this_image()
  np = num_images()
  scalar = -1

  if(me /= 1) call sleep(1)

  scalar = 1

  sync all

  if(me == 1) then
     do i=1,np
        if(scalar[i] /= 1) then
           write(*,*) 'Test failed.'
           stop
        endif
     end do
  end if

  if(me == 1) print *,'Test passed.'

end program syncall
